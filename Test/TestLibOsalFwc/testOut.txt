63: void testReadLine()
64: {
65:   char sbuffer[50];  //small buffer for line number
66:   int ibuffer;      //current number of chars in buffer
67:   ReadLine_Fwc* fin = (ReadLine_Fwc*)os_allocMem(sizeof(ReadLine_Fwc));  //use malloc because it may be too much in stack.
68:   bool bOk = open_ReadLine_Fwc(fin, "src/File_TestLibOsalFwc.c");
69:   OS_HandleFile fout = os_fopenToWrite("testOut.txt", false);
70:   char const* sLine;
71:   int lineNumber = 1;
72:   while(bOk && (sLine = readLine_ReadLine_Fwc(fin))!=null){
73:     //prepare the line number. We havn't a printf at osal level.
74:     ibuffer = format_Formatter_FW(null, "%d: ", 4, sbuffer, sizeof(sbuffer), "I", lineNumber);
75:     os_fwrite(fout, sbuffer, ibuffer);
76:     //write the read line:
77:     os_fwrite(fout, sLine, strlen_Fwc(sLine, 1000));
78:     os_fwrite(fout, "\n", 1);
79:     //
80:     lineNumber +=1;
81:   }
82:   os_fclose(fout);
83: }
